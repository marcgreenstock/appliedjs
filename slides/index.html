The build process

How do we build a client-side JavasSript app without a backend?

Say hello to Gulp

A Simple task

            
var gulp = require('gulp');

gulp.task('copy', function() {
  gulp
    .src('./src/**/*')
    .pipe(gulp.dest('./_build'));
});

Building styles

            
var less = require('gulp-less');

gulp.task('styles', function() {
  gulp
    .src('./src/assets/styles/app.less')
    .pipe(less())
    .pipe(gulp.dest('./_build/assets'));
});

Building JavaScript

The tool belt

          	
var browserify = require('browserify');
var reactify = require('coffee-reactify');
var envify = require('envify/custom');
var coffeeify = require('coffeeify');
var source = require('vinyl-source-stream');

The gulp task

          	
gulp.task('scripts', function() {
  var options = {
    entries: './src/assets/scripts/app.coffee',
    extensions: ['.cjsx', '.coffee', '.litcoffee'],
    debug: true
  };
  browserify(options)
    .transform(reactify, {
      coffeeout: true
    })
    .transform(coffeeify)
    .transform(envify, {
      apiUrl: 'https://api.example.com',
      lastModified: (new Date).toUTCString()
    })
    .bundle()
    .pipe(source('app.js'))
    .pipe(gulp.dest('./_build/assets');
});

The tool belt

						
var revall = require('gulp-rev-all');
var awspublish = require('gulp-awspublish');
var cloudfront = require('gulp-cloudfront');

The gulp task

						
gulp.task('publish', ['styles', 'scripts'], function() {
  var aws = require('./credentials').production.aws;
  var headers = {
    'Cache-Control': 'max-age=315360000, no-transform, public'
  };
  var publisher = awspublish.create(aws);

  gulp.src('./_build/**')
    .pipe(revall())
    .pipe(awspublish.gzip())
    .pipe(publisher.publish(headers))
    .pipe(publisher.cache())
    .pipe(awspublish.reporter())
    .pipe(cloudfront(aws))
});

Development

Watching for changes

						
var watch = require('gulp-watch');

gulp.task('watch', function(callback) {
  watch './src/assets/scripts/**/*', function() {
    gulp.start('scripts');
  });
  watch './src/assets/styles/**/*', function() {
    gulp.start('styles');
  });
  callback();
});

Local hosting

						
var webserver = require('gulp-webserver');

gulp.task('serve', function() {
  gulp.src('./_build')
    .pipe(webserver({
      livereload: false,
      open: false
    });
});
